$("button#choose-table-seat").click(function(e){
var formValid = false;
if ( $("form.choose-table-seat input:checkbox:checked").length > 0 ) {
if ( $("form.choose-table-seat input[name='ticket_section_name']").val() == 'table' ) {
formValid = true;
}else{

$("form.choose-table-seat input:checkbox:checked").each(function()
{
if ( $("form.choose-table-seat input[name='ticket_class_class']").val() == "after-party") {
var qty = $("form.choose-table-seat input[name='choose-ticket-quantity-"+$(this).val()+"']").val();
var maxQty = $("form.choose-table-seat input[name='choose-ticket-quantity-"+$(this).val()+"']").attr('max');
}else{
var qty = $("form.choose-table-seat select[name='choose-ticket-quantity-"+$(this).val()+"']").val();
var maxQty = $("form.choose-table-seat select[name='choose-ticket-quantity-"+$(this).val()+"']").attr('max');
}
if ( $.trim(qty) == "" || parseInt(qty) < 1 ) {
formValid = false;
return;
}else if ( parseInt(qty) > parseInt(maxQty) ) {
$('.alert-error-seat-select').html("<strong>Error!</strong> Please correct table/seat quantity.");
formValid = false;
return;
}else if ( parseInt(qty) > 0 ) {
formValid = true;
}else{
formValid = false;
}
});
}
}

if (formValid) {
$.ajax({
type: "POST",
url: "http://cluster1.website-staging.uk/ticketbaby.co.uk/index.php/ajax/setSessionCart",
data: $('form.choose-table-seat').serialize(),
success: function(msg){
loadSessionCart();
$('#ticketSelect').modal('hide');
$('html, body').animate({
scrollTop: ($('.session-cart-list').offset().top)
},1500);
},
error: function(){
$('#ticketSelect').modal('hide');
}
});
}else{
$('.alert-error-seat-select').show();
}

});

}
});
